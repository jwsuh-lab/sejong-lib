| 실시간 스트리밍 | **SSE (Server-Sent Events)** | 단방향 충분, WebSocket 대비 단순 |

### 3.3 status enum 통합 (프론트 ↔ 백엔드)

백엔드 DB의 4단계 status를 프론트 3단계로 매핑한다.

| 백엔드 (DB) | 프론트 (UI) | 설명 |
|-------------|------------|------|
| `collected` | `unverified` | 크롤링만 완료, 미선별 |
| `selected` | `unverified` | 선별됨, 검증 대기 |
| `verified` | `verified` | 3중 검증 통과 |
| `excluded` | `excluded` | 제외 처리 |

```python
# api/schemas.py
FRONT_STATUS_MAP = {
    "collected": "unverified",
    "selected": "unverified",
    "verified": "verified",
    "excluded": "excluded",
}

class DocumentResponse(BaseModel):
    status: Literal["verified", "unverified", "excluded"]

    @model_validator(mode="before")
    @classmethod
    def map_status(cls, data):
        if isinstance(data, dict) and "status" in data:
            data["status"] = FRONT_STATUS_MAP.get(data["status"], data["status"])
        return data
```

  ```python
  def get_background_session():
      """백그라운드 태스크(PipelineRunner 등)용 독립 DB 세션"""
      session = SessionLocal()

          yield session

          session.close()
  ```

### Phase 3: 파이프라인 실행 API (4일)

#### Task 3.1 — Pipeline Runner 서비스

기존 Python 스크립트를 **subprocess로 호출**하되, stdout을 캡처하여 SSE로 스트리밍한다.
크롤링 대상 스크립트 목록은 **config.json에서 동적 로드**하여, 하드코딩 없이 관리한다.

```python
# api/services/pipeline_runner.py
class PipelineRunner:
    """기존 스크립트를 subprocess로 실행하고 로그를 스트리밍"""

    _current_proc: asyncio.subprocess.Process | None = None

    def __init__(self, broadcaster: LogBroadcaster, sync_service: SyncService):
        self.broadcaster = broadcaster
        self.sync_service = sync_service
        self._load_steps()

    # config.json 불존재 시 사용할 기본 크롤링 스크립트
    DEFAULT_CRAWL_SCRIPTS = [
        {"script": "main.py", "args": ["crawl-us", "--all"], "label": "미국"},
        {"script": "main.py", "args": ["crawl-se", "--all"], "label": "스웨덴"},
        {"script": "main.py", "args": ["crawl-sg", "--all"], "label": "싱가포르"},
        {"script": "crawl_additional.py", "args": [], "label": "추가소스"},
        {"script": "crawl_eprs.py", "args": [], "label": "EPRS"},
        {"script": "crawl_more.py", "args": [], "label": "추가소스2"},
        {"script": "crawl_new_sites.py", "args": [], "label": "신규사이트"},
    ]

    def _load_steps(self):
        """config.json에서 크롤링 스크립트 목록 동적 로드 (불존재 시 기본값 폴백)"""
        try:
            config = json.loads(Path("config.json").read_text())
        except (FileNotFoundError, json.JSONDecodeError):
            config = {}  # ★ 최초 설치 시에도 API 서버 정상 시작
        crawl_scripts = [
            (s["script"], s["args"])
            for s in config.get("crawl_scripts", self.DEFAULT_CRAWL_SCRIPTS)
            if s["script"] not in config.get("crawl_exclude", [])
        ]
        self.STEPS = {
            1: {"name": "크롤링", "scripts": crawl_scripts, "fail_policy": "continue"},
            2: {"name": "선별",   "scripts": [("select_300.py", [])], "fail_policy": "abort"},
            3: {"name": "보강",   "scripts": [
                    ("resolve_dates_v2.py", []),
                    ("resolve_pdfs.py", []),
                    ("resolve_authors.py", []),
                    ("resolve_excerpts.py", []),
                    ("enrich_selection.py", []),
                    ("metadata_resolver.py", []),
                ], "fail_policy": "continue"},
            4: {"name": "검증", "scripts": [("validate_selection.py", [])], "fail_policy": "abort"},
            5: {"name": "출력", "scripts": [("export_selection.py", [])], "fail_policy": "abort"},
        }

    async def run_step(self, step: int, run_id: int) -> str:
        """단일 단계 실행 — stdout을 줄 단위로 캡처"""
        step_info = self.STEPS[step]
        has_failure = False

        for script, args in step_info["scripts"]:
            data_dir = str(Path(BACKEND_DIR) / "data")
            full_args = [sys.executable, script] + args
            # --data-dir 인자가 있는 스크립트에 절대경로 전달
            if "--data-dir" not in args:
                full_args += ["--data-dir", data_dir] if self._supports_data_dir(script) else []

            self._current_proc = await asyncio.create_subprocess_exec(
                *full_args,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.STDOUT,
                cwd=BACKEND_DIR,
            )
            async for line in self._current_proc.stdout:
                text = line.decode().strip()
                if text:
                    await self._emit_log(run_id, step, text)
            await self._current_proc.wait()

            if self._current_proc.returncode != 0:
                await self._emit_log(run_id, step,
                    f"ERROR: {script} 실패 (exit code={self._current_proc.returncode})")
                if step_info["fail_policy"] == "abort":
                    raise PipelineStepError(step, script)
                else:
                    has_failure = True

            self._current_proc = None

        # 단계 완료 후 JSON → DB 동기화
        await self.sync_service.sync_after_step(step)
        return "partial" if has_failure else "completed"

    async def stop(self):
        """실행 중인 프로세스를 안전하게 종료"""
        if self._current_proc and self._current_proc.returncode is None:
            self._current_proc.terminate()
            try:
                await asyncio.wait_for(self._current_proc.wait(), timeout=10)
            except asyncio.TimeoutError:
                self._current_proc.kill()  # 10초 후 강제 종료
            self._current_proc = None

    async def _emit_log(self, run_id: int, step: int, text: str):
        level = self._parse_level(text)
        log_entry = {"run_id": run_id, "step": step, "message": text,
                     "level": level, "timestamp": datetime.now().isoformat()}
        # ★ 백그라운드 태스크 전용 세션 사용 (FastAPI Depends 스코프 밖)
        with get_background_session() as session:
            db_log = PipelineLog(**log_entry)
            session.add(db_log)
            session.flush()           # id 확보 (SSE id: 필드용)
            log_entry["id"] = db_log.id
        # SSE 브로드캐스트 (큐 기반, 폴링 없음)
        await self.broadcaster.broadcast(log_entry)
```

#### Task 3.2 — SSE 로그 브로드캐스팅 (asyncio.Queue 기반)

DB 폴링 대신 인메모리 큐 기반 브로드캐스트를 사용한다.
DB 저장은 영구 이력용으로 유지하되, SSE 스트리밍은 큐에서 즉시 전달한다.

```python
# api/services/log_broadcaster.py
class LogBroadcaster:
    """다중 SSE 클라이언트에 로그를 브로드캐스트 (메모리 보호 포함)"""

    def __init__(self):
        self._subscribers: list[asyncio.Queue] = []

    def subscribe(self) -> asyncio.Queue:
        q: asyncio.Queue = asyncio.Queue(maxsize=1000)  # ★ 메모리 보호
        self._subscribers.append(q)
        return q

    def unsubscribe(self, q: asyncio.Queue):
        if q in self._subscribers:
            self._subscribers.remove(q)

    async def broadcast(self, log: dict):
        for q in self._subscribers:
            if q.full():
                try:
                    q.get_nowait()  # 오래된 로그 드롭
                except asyncio.QueueEmpty:
                    pass
            try:
                q.put_nowait(log)  # 논블로킹 — 느린 subscriber에 의한 블로킹 방지
            except asyncio.QueueFull:
                pass

# api/routers/pipeline.py
@router.get("/logs")
async def stream_logs(request: Request):
    # ★ Last-Event-Id 헤더로 재연결 시 누락 로그 복구
    last_id = int(request.headers.get("Last-Event-Id", "0"))
    queue = broadcaster.subscribe()

    async def event_generator():
        try:
            # Phase 1: 재연결 시 DB에서 누락분 먼저 전송
            if last_id > 0:
                with get_background_session() as session:
                    missed = session.query(PipelineLog).filter(
                        PipelineLog.id > last_id
                    ).order_by(PipelineLog.id).all()
                    for log in missed:
                        yield f"id: {log.id}\ndata: {json.dumps(log.to_dict())}\n\n"

            # Phase 2: 큐 스트리밍으로 전환
            while True:
                if await request.is_disconnected():
                    break
                try:
                    log = await asyncio.wait_for(queue.get(), timeout=30)
                    yield f"id: {log.get('id', '')}\ndata: {json.dumps(log)}\n\n"
                except asyncio.TimeoutError:
                    yield f": keepalive\n\n"  # SSE 연결 유지
        finally:
            broadcaster.unsubscribe(queue)

    return StreamingResponse(event_generator(), media_type="text/event-stream")
```

#### Task 3.3 — JSON ↔ DB 동기화 서비스

파이프라인 실행 후 JSON 파일이 갱신되면 DB에 반영한다.
UPSERT 키는 `(site_code, link)` 복합 인덱스를 사용한다.

```python
# api/services/sync_service.py
class SyncService:
    async def sync_after_step(self, step: int):
        """단계 완료 후 JSON → DB 동기화"""
        if step == 1:  # 크롤링
            await self._import_crawl_jsons()
        elif step == 2:  # 선별
            await self._import_selection("selection_500.json", target_status="selected")
        elif step in (3, 4):  # 보강/검증
            await self._update_selection("selection_400.json")
        elif step == 5:  # 출력
            pass  # Excel 생성만, DB 변경 없음

    async def _import_crawl_jsons(self):
        """data/*.json 신규 파일 → documents 테이블 UPSERT"""
        for path in DATA_DIR.glob("*_Z*_*.json"):
            data = json.loads(path.read_text())
            for doc in data.get("documents", []):
                # (site_code, link) 기준 UPSERT
                existing = db.query(Document).filter(
                    Document.site_code == doc.get("site_code"),
                    Document.link == doc.get("link"),
                ).first()
                if existing:
                    for k, v in doc.items():
                        if v is not None:
                            setattr(existing, k, v)
                else:
                    db.add(Document(**doc, source_file=path.name))
        db.commit()

    async def _update_selection(self, filename: str):
        """selection JSON → documents 테이블 UPDATE (검증/보강 필드)"""
        selection = json.loads((DATA_DIR / filename).read_text())
        for doc in selection:
            db.execute(
                update(Document)
                .where(Document.site_code == doc.get("_site_code", doc.get("site_code")))
                .where(Document.link == doc.get("link"))
                .values(
                    relevance_score=doc.get("_relevance_score"),
                    status="verified" if doc.get("_validated") else
                           "excluded" if doc.get("_excluded") else "selected",
                    excluded=doc.get("_excluded", False),
                    exclude_reasons=json.dumps(doc.get("_exclude_reasons", [])),
                    date_verified=doc.get("_date_verified", False),
                    date_verified_by=doc.get("_date_verified_by"),
                    pdf_verified=doc.get("_pdf_ok", False),
                    keywords=doc.get("keywords"),
                    journal=doc.get("journal"),
                    isbn=doc.get("isbn"),
                    issn=doc.get("issn"),
                    license=doc.get("license"),
                    updated_at=datetime.now(),
                )
            )
        db.commit()
```

#### Task 3.5 — Atomic JSON Write 유틸리티

기존 스크립트의 JSON 파일 쓰기를 안전하게 보호한다.
파이프라인 중지 시 데이터 손상을 방지한다.

```python
# utils/safe_json.py
import tempfile, os, json

def atomic_json_write(path: str, data, ensure_ascii=False, indent=2):
    """임시파일에 쓴 후 rename — 중단 시에도 원본 보존"""
    dir_name = os.path.dirname(os.path.abspath(path))
    with tempfile.NamedTemporaryFile('w', dir=dir_name, suffix='.tmp',
                                      delete=False, encoding='utf-8') as f:
        json.dump(data, f, ensure_ascii=ensure_ascii, indent=indent)
        tmp_path = f.name
    os.replace(tmp_path, path)  # atomic on POSIX
```

## 8. 위험 요소 및 대응

| 위험 | 영향 | 대응 |
|------|------|------|
| 기존 스크립트 stdout 형식 비표준 | 로그 파싱 실패 | 정규식 기반 유연한 파서 + 원본 그대로 저장 |
| selection_400.json in-place 수정 | 동시 접근 충돌 | **atomic write** + 파이프라인 실행 중 DB 쓰기 잠금 + 완료 후 동기화 |
| 장시간 크롤링 (30분+) | SSE 연결 끊김 | **keepalive 전송** + EventSource 자동 재연결 + DB에 로그 영구 저장 |
| SQLite 동시 쓰기 제한 | 동시 요청 시 잠금 | **WAL 모드 활성화** + 쓰기 작업 큐잉 |
| 기존 CLI 사용자 혼란 | 워크플로우 중복 | CLI/웹 둘 다 지원, DB 동기화로 일관성 보장 |
| 크롤링 스크립트 data_dir 경로 차이 | subprocess에서 상대경로 참조 실패 | **cwd + --data-dir 절대경로** 명시적 전달 |
| sejong.db Git 충돌 | DB 파일이 커밋됨 | **.gitignore에 sejong.db* 추가** |
| CORS 보안 | allow_origins=["*"] 위험 | **환경변수 CORS_ORIGINS** 기반, 기본값 localhost만 |
| selection 파일명 혼재 | 동기화 시 어느 파일이 최신인지 불명확 | **파일별 용도 명세 (섹션 3.5)** + 단계별 대상 파일 명확 정의 |
| 파이프라인 중지 시 데이터 손상 | JSON 파일 쓰기 중 terminate | **atomic write (임시파일 → rename)** + 10초 graceful shutdown |
| 에러 처리 미정의 | 스크립트 실패 시 동작 불명 | **fail_policy (continue/abort)** 단계별 정의 |
